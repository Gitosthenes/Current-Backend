ENDPOINT

/send post authentication: JWT

sender: int,

receiver: int

 JSON

 JSON

sender: int,

receiver: int,

accepted: string "true/false"

 JSON

success: boolean

memberid: int

 JSON

memberid: int, firstname: string, lastname: string, username: string, email: string, verified: int, sender: int

memberid: int,

firstname: string,

lastname: string

username: string
